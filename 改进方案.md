# 小程序项目改进方案

## 📋 项目概述

这是一个微信小程序项目，包含4个主要页面：
- **index** - 首页/测试页面
- **GEO** - 工作流聊天页面
- **Agent** - 智能体聊天页面
- **Game** - 游戏页面

## 🔴 主要问题

### 1. 大量代码重复

**问题描述：**
- WebSocket连接、心跳、重连机制在 Agent.js、GEO.js、Game.js 中几乎完全相同（约300行重复代码）
- 头像相关方法在3个页面中完全重复（约200行重复代码）
- 历史记录管理逻辑高度相似（约150行重复代码）
- 日志、时间格式化等工具函数在每个页面都重复定义

**影响：**
- 代码维护成本高
- 修改需要在多处同步
- 容易出现不一致

### 2. 安全隐患 ⚠️

class ErrorHandler {
  static handle(error, context = '') {
    console.error(`[${context}]`, error);

    let message = '操作失败，请稍后重试';

    if (error.errMsg) {
      if (error.errMsg.includes('network')) {

      } else if (error.errMsg.includes('timeout')) {
        message = '请求超时，请重试';

    }

    wx.showToast({
      title: message,
      icon: 'none',
      duration: 3000
    });

    // 可以添加错误上报
    this.report(error, context);
  }

  static report(error, context) {
    // 错误上报逻辑

  }
}

module.exports = ErrorHandler;

// store/index.js
class Store {
  constructor() {
    this.state = {
      user: {
        id: '101',
        preferences: {}
      },
      app: {
        isConnected: false
      }
    };
    this.listeners = [];
  }

  getState() {
    return this.state;
  }

  setState(path, value) {
    // 简单的状态更新
    const keys = path.split('.');
    let obj = this.state;
    for (let i = 0; i < keys.length - 1; i++) {
      obj = obj[keys[i]];
    }
    obj[keys[keys.length - 1]] = value;

    this.notify();
  }

  subscribe(listener) {
    this.listeners.push(listener);
    return () => {
      const index = this.listeners.indexOf(listener);
      this.listeners.splice(index, 1);
    };
  }

  notify() {
    this.listeners.forEach(listener => listener(this.state));
  }
}

module.exports = new Store();

// utils/performance.js
function debounce(fn, delay = 300) {
  let timer = null;
  return function(...args) {
    if (timer) clearTimeout(timer);
    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

function throttle(fn, delay = 300) {
  let last = 0;
  return function(...args) {
    const now = Date.now();
    if (now - last > delay) {
      last = now;
      fn.apply(this, args);
    }
  };
}

module.exports = { debounce, throttle };

```javascript
// utils/websocket.js
class WebSocketManager {
  constructor(options = {}) {
    this.url = options.url;
    this.heartbeatInterval = options.heartbeatInterval || 30000;
    this.maxReconnectCount = options.maxReconnectCount || 5;

    this.socketTask = null;
    this.isConnected = false;
    this.reconnectCount = 0;
    this.heartbeatTimer = null;
    this.reconnectTimer = null;
    this.messageHandlers = [];
  }

  connect() {
    return new Promise((resolve, reject) => {
      this.socketTask = wx.connectSocket({
        url: this.url,
        success: () => {
          this.setupEventListeners();
          resolve();
        },
        fail: reject
      });
    });
  }

  setupEventListeners() {
    this.socketTask.onOpen(() => {
      this.isConnected = true;
      this.reconnectCount = 0;
      this.startHeartbeat();
      this.emit('open');
    });

    this.socketTask.onMessage((res) => {
      this.emit('message', res);
    });

    this.socketTask.onClose(() => {
      this.isConnected = false;
      this.stopHeartbeat();
      this.tryReconnect();
      this.emit('close');
    });

    this.socketTask.onError((err) => {
      this.emit('error', err);
    });
  }

  send(data) {
    if (!this.isConnected) {
      return Promise.reject(new Error('WebSocket未连接'));
    }

    return new Promise((resolve, reject) => {
      this.socketTask.send({
        data: JSON.stringify(data),
        success: resolve,
        fail: reject
      });
    });
  }

  on(event, handler) {
    this.messageHandlers.push({ event, handler });
  }

  off(event, handler) {
    this.messageHandlers = this.messageHandlers.filter(
      h => h.event !== event || h.handler !== handler
    );
  }

  emit(event, data) {
    this.messageHandlers
      .filter(h => h.event === event)
      .forEach(h => h.handler(data));
  }

  startHeartbeat() {
    this.stopHeartbeat();
    this.heartbeatTimer = setInterval(() => {
      if (this.isConnected) {
        this.send({ type: 'ping', timestamp: Date.now() });
      }
    }, this.heartbeatInterval);
  }

  stopHeartbeat() {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }

  tryReconnect() {
    if (this.reconnectCount >= this.maxReconnectCount) {
      this.emit('maxReconnectReached');
      return;
    }

    this.reconnectCount++;
    const delay = Math.min(1000 * Math.pow(2, this.reconnectCount), 30000);

    this.reconnectTimer = setTimeout(() => {
      this.connect();
    }, delay);
  }

  disconnect() {
    this.stopHeartbeat();
    if (this.reconnectTimer) {
      clearTimeout(this.reconnectTimer);
    }
    if (this.socketTask) {
      this.socketTask.close();
    }
    this.isConnected = false;
  }
}

module.exports = WebSocketManager;
```
